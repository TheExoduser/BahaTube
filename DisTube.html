    getQueue(message) → {Queue}

    Get the guild queue

    Parameters:
    Name Type Description
    message Discord.Message

    The message from guild channel5643
    Returns:

    The guild queue

    Type
    Queue
    Example 
    client.on('message', (message) => {
    if (!message.content.startsWith(config.prefix)) return;
    const args = message.content.slice(config.prefix.length).trim().split(/ +/g);
    const command = args.shift();
    if (command == "queue") {
        let queue = distube.getQueue(message);
        message.channel.send('Current queue:\n' + queue.songs.map((song, id) =>
            `**${id+1}**. [${song.name}](${song.url}) - \`${song.formattedDuration}\``
        ).join("\n"));
    }
});

    setFilter(message, filter) → {string}

    @2.0.0 Enable or disable a filter of the queue, replay the playing song. Available filters: Filter

    Parameters:
    Name Type Description
    message Discord.Message

    The message from guild channel
    filter Filter

    A filter name
    Returns:

    Current queue's filter name.

    Type
    string
    Example 
    client.on('message', (message) => {
    if (!message.content.startsWith(config.prefix)) return;
    const args = message.content.slice(config.prefix.length).trim().split(/ +/g);
    const command = args.shift();
    if ([`3d`, `bassboost`, `echo`, `karaoke`, `nightcore`, `vaporwave`].includes(command)) {
        let filter = distube.setFilter(message, command);
        message.channel.send("Current queue filter: " + (filter || "Off"));
    }
});

    setRepeatMode(message, mode) → {number}

    Set the repeat mode of the guild queue. Turn off if repeat mode is the same value as new mode. Toggle mode: mode = null (0 -> 1 -> 2 -> 0...)

    Parameters:
    Name Type Default Description
    message Discord.Message

    The message from guild channel
    mode number null

    The repeat modes (0: disabled, 1: Repeat a song, 2: Repeat all the queue)
    Returns:

    The new repeat mode

    Type
    number
    Example 
    client.on('message', (message) => {
    if (!message.content.startsWith(config.prefix)) return;
    const args = message.content.slice(config.prefix.length).trim().split(/ +/g);
    const command = args.shift();
    if (command == "repeat") {
        let mode = distube.setRepeatMode(message, parseInt(args[0]));
        mode = mode ? mode == 2 ? "Repeat queue" : "Repeat song" : "Off";
        message.channel.send("Set repeat mode to `" + mode + "`");
    }
});

    searchResult

    Emitted when DisTubeOptions.searchSongs is true. DisTube will wait for user's next message to choose song manually if song param of DisTube#play() is invalid url

    Parameters:
    Name Type Description
    message Discord.Message

    The message from guild channel
    result Array.<Song>

    Searched result (max length = 12)
    Example 
    // DisTubeOptions.searchSongs = true
distube.on("searchResult", (message, result) => {
    let i = 0;
    message.channel.send(`**Choose an option from below**\n${result.map(song => `**${++i}**. ${song.title} - \`${song.duration}\``).join("\n")}\n*Enter anything else or wait 60 seconds to cancel*`);
});
